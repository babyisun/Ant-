# Copy

复制到粘贴板。

## 代码演示

简单示例

<Playground>
  {
    () => {
        class Demo extends Component {
            constructor() {
                this.state = {
                    value: ''
                }
            }
            render() {
                const { value } = this.state
                return (
                <Fragment>
                    <Row gutter={16}>
                        <Col span={18}>
                            <Input onChange={e => this.setState({value: e.target.value})}/>
                        </Col>
                        <Col span={6}>
                            <Copy text={value} type="text"/>
                        </Col>
                    </Row>
                </Fragment>
            )
            }
        }
        return <Demo/>
     }
}
</Playground>
